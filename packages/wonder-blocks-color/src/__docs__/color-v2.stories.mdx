## Usage

export const SwatchItem = ({children, foreground, background, inverted}) => (
    <View
        style={{
            color: !inverted ? foreground : background,
            backgroundColor: !inverted ? background : foreground,
            gap: Spacing.medium_16,
            flex: 1,
            justifyContent: "center",
            alignItems: "center",
            padding: Spacing.medium_16,
        }}
    >
        <LabelLarge>{children}</LabelLarge>
    </View>
);

export const Swatch = ({name, value, use, description, children}) => (
    <View
        style={{
            flexDirection: "row",
            gap: Spacing.xSmall_8,
            marginBottom: Spacing.xLarge_32,
        }}
    >
        <View
            style={{
                flexBasis: "30%",
            }}
        >
            <View
                style={{flexDirection: "row", justifyContent: "space-between"}}
            >
                <LabelLarge>{name}</LabelLarge>
                <LabelXSmall
                    style={{
                        background: Color.offBlack8,
                        border: `solid 1px ${Color.offBlack16}`,
                        borderRadius: Spacing.xxxxSmall_2,
                        marginLeft: Spacing.xxxSmall_4,
                        marginRight: "auto",
                        padding: `0 ${Spacing.xxxSmall_4}px`,
                    }}
                >
                    {value}
                </LabelXSmall>
                {use ? <LabelSmall style={{fontWeight: "bold"}}>{use}</LabelSmall> : null}
            </View>
            <Caption>{description}</Caption>
        </View>
        <View
            style={{
                flexDirection: "row",
                flexBasis: "70%",
                borderRadius: 4,
                background: Color.white,
                boxShadow: `0 1px 3px 0 ${Color.offBlack8}`,
                border: `1px solid ${Color.offBlack16}`,
                overflow: "hidden",
                height: Spacing.xxLarge_48,
            }}
        >
            {children}
        </View>
    </View>
);
